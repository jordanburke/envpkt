import { Card, CardGrid } from "@astrojs/starlight/components"

<CardGrid stagger>
  <Card title="MCP Integration" icon="puzzle">
    Built-in MCP server gives AI agents structured awareness of their credentials — capabilities, constraints,
    expiration, rotation — without exposing secret values. Works with Claude, Cursor, VS Code, and any MCP client.
  </Card>
  <Card title="Three-Tier Security" icon="approve-check">
    Architecturally enforced trust boundaries. The MCP server never has access to secret values. Runtime injection stays
    outside the LLM context. Fleet-wide audit trails for full visibility.
  </Card>
  <Card title="Metadata + Sealed Secrets" icon="information">
    Every credential gets structured metadata — service, purpose, capabilities, expiration, rotation URL — plus optional
    age-encrypted sealed packets, safe to commit to git.
  </Card>
  <Card title="Environment Scanning" icon="magnifier">
    Auto-discover credentials from your shell with `envpkt env scan`. Matches ~45 known services, ~13 suffix patterns,
    and ~29 value shapes with confidence scoring.
  </Card>
  <Card title="Fleet Health" icon="heart">
    Scan an entire directory tree of agents with `envpkt fleet`. Get aggregated health status, expiration warnings, and
    stale credential detection across your fleet.
  </Card>
  <Card title="Functional API" icon="seti:typescript">
    TypeScript library built on functype. All functions return `Either` or `Option` — no thrown exceptions. Use
    programmatically for boot, audit, fleet scan, and more.
  </Card>
</CardGrid>
